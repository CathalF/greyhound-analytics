for bet in value_bets:
    if bet.get('factor_breakdown'):
        # Find highest contributing factor (excluding base)
        factors = bet['factor_breakdown']
        highest = max(
            [(k, v['value']) for k, v in factors.items() if k != 'base_value'],
            key=lambda x: abs(x[1] - 1.0),
