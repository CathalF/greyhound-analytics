<task type="checkpoint:decision" gate="blocking">
  <decision>Select dog data access strategy</decision>
  <context>Based on manual exploration, choose the best approach for systematically collecting dog statistics.</context>
  <options>
    <option id="search-based">
      <name>Search-Based Access</name>
      <pros>Direct access to specific dogs, can target active/popular dogs, respects site structure</pros>
      <cons>Need list of dog names to search, slower (one request per dog)</cons>
    </option>
    <option id="listing-crawl">
      <name>Listing Page Crawl</name>
      <pros>Discover dogs automatically via top dogs / track listings, efficient batch discovery</pros>
      <cons>May miss some dogs, depends on site having good listing pages</cons>
    </option>
    <option id="hybrid">
      <name>Hybrid Approach</name>
      <pros>Use listings to discover popular dogs, search for specific dogs when needed</pros>
      <cons>More complex, two code paths to maintain</cons>
    </option>
  </options>
  <resume-signal>Select: search-based, listing-crawl, or hybrid</resume-signal>
</task>
